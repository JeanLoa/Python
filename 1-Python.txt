num1 = int(input("Ingresa el primer numero: "))
num2 = int(input("Ingresa el segundo numero: "))

divisionentera = num1 // num2
resto = num1 % num2
potencia = num1**num2

print(f"La division entera entre {num1} y {num2} es: {divisionentera}")
print(f"El resto de la division entre {num1} y {num2} es: {resto}")
print(f"La potencia de base {num1} a {num2} es: {potencia}")
